        Q2 Write code to implement timer clock using JS -- display HH:MM:SS -- the time should keep updating every second
        ans ->
        function setTimeInBox(){
            let date=new Date();
            if(date.getHours()>12){
                box[0].innerText=date.getHours()-12;
            }
            else{ 
                box[0].innerText=date.getHours();
